```yaml
# compliance-policies.rego
# Open Policy Agent (OPA) policies

package kubernetes.admission

# Deny containers running as root
deny[msg] {
  input.request.kind.kind == "Pod"
  container := input.request.object.spec.containers[_]
  container.securityContext.runAsUser == 0
  msg := sprintf("Container %v is running as root", [container.name])
}

# Require resource limits
deny[msg] {
  input.request.kind.kind == "Pod"
  container := input.request.object.spec.containers[_]
  not container.resources.limits
  msg := sprintf("Container %v missing resource limits", [container.name])
}

# Enforce namespace labels
deny[msg] {
  input.request.kind.kind == "Namespace"
  not input.request.object.metadata.labels.owner
  msg := "Namespace missing required 'owner' label"
}
```

```yaml
# .github/workflows/compliance-scan.yml

name: Compliance Scanning

on:
  push:
    branches: [main]
  pull_request:

jobs:
  terraform-compliance:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Run Checkov scan
        uses: bridgecrewio/checkov-action@master
        with:
          directory: terraform/
          framework: terraform
          output_format: sarif
          soft_fail: false
          skip_check: CKV_AWS_20  # Skip specific checks if needed

      - name: Compliance report
        run: |
          echo "## Compliance Scan Results" >> $GITHUB_STEP_SUMMARY
          echo "Scan completed. Review SARIF output for details." >> $GITHUB_STEP_SUMMARY
```

Save to: `deliverables/compliance_report.md`
