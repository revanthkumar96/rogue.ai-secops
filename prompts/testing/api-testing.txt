"""
API Test Suite for {{WEB_URL}}
Tests REST API endpoints with comprehensive coverage

Coverage:
- CRUD operations for all resources
- Authentication and authorization
- Error handling
- Response validation
"""

import pytest
import requests
from typing import Dict, Any


# ============================================
# Configuration
# ============================================

API_BASE_URL = "{{WEB_URL}}/api"
API_TIMEOUT = 30  # seconds


# ============================================
# Fixtures
# ============================================

@pytest.fixture(scope="session")
def api_client():
    """Fixture to provide API client session"""
    session = requests.Session()
    session.headers.update({"Content-Type": "application/json"})
    yield session
    session.close()


@pytest.fixture(scope="function")
def auth_token(api_client) -> str:
    """Fixture to provide authentication token"""
    # Login to get auth token
    response = api_client.post(
        f"{API_BASE_URL}/auth/login",
        json={"username": "test_user", "password": "Test123!"}
    )
    assert response.status_code == 200
    return response.json()["token"]


@pytest.fixture(scope="function")
def auth_headers(auth_token) -> Dict[str, str]:
    """Fixture to provide headers with authentication"""
    return {"Authorization": f"Bearer {auth_token}"}


# ============================================
# Test Suites
# ============================================

class TestUserAPI:
    """Test suite for User API endpoints"""

    def test_create_user_success(self, api_client):
        """Test creating a new user with valid data"""
        # Arrange
        user_data = {
            "username": f"test_user_{pytest.timestamp}",
            "email": f"test_{pytest.timestamp}@example.com",
            "password": "SecurePass123!"
        }

        # Act
        response = api_client.post(f"{API_BASE_URL}/users", json=user_data)

        # Assert
        assert response.status_code == 201
        response_data = response.json()
        assert response_data["username"] == user_data["username"]
        assert response_data["email"] == user_data["email"]
        assert "password" not in response_data  # Password should not be returned
        assert "id" in response_data
        assert response.headers["Content-Type"] == "application/json"

    def test_create_user_duplicate_email(self, api_client):
        """Test creating user with duplicate email returns error"""
        # Arrange
        user_data = {
            "username": "test_user_dup",
            "email": "duplicate@example.com",
            "password": "SecurePass123!"
        }
        # Create user first time
        api_client.post(f"{API_BASE_URL}/users", json=user_data)

        # Act - try to create again
        response = api_client.post(f"{API_BASE_URL}/users", json=user_data)

        # Assert
        assert response.status_code == 409  # Conflict
        assert "already exists" in response.json()["error"].lower()

    def test_get_user_by_id(self, api_client, auth_headers):
        """Test retrieving user by ID"""
        # Arrange - create user first
        user_data = {"username": "get_test", "email": "get@test.com", "password": "Pass123!"}
        create_response = api_client.post(f"{API_BASE_URL}/users", json=user_data)
        user_id = create_response.json()["id"]

        # Act
        response = api_client.get(f"{API_BASE_URL}/users/{user_id}", headers=auth_headers)

        # Assert
        assert response.status_code == 200
        user = response.json()
        assert user["id"] == user_id
        assert user["username"] == user_data["username"]

    def test_update_user(self, api_client, auth_headers):
        """Test updating user information"""
        # Arrange - create user
        create_response = api_client.post(
            f"{API_BASE_URL}/users",
            json={"username": "update_test", "email": "update@test.com", "password": "Pass123!"}
        )
        user_id = create_response.json()["id"]
        update_data = {"email": "newemail@test.com"}

        # Act
        response = api_client.patch(
            f"{API_BASE_URL}/users/{user_id}",
            json=update_data,
            headers=auth_headers
        )

        # Assert
        assert response.status_code == 200
        assert response.json()["email"] == update_data["email"]

    def test_delete_user(self, api_client, auth_headers):
        """Test deleting a user"""
        # Arrange - create user
        create_response = api_client.post(
            f"{API_BASE_URL}/users",
            json={"username": "delete_test", "email": "delete@test.com", "password": "Pass123!"}
        )
        user_id = create_response.json()["id"]

        # Act
        response = api_client.delete(f"{API_BASE_URL}/users/{user_id}", headers=auth_headers)

        # Assert
        assert response.status_code == 204
        # Verify user is deleted
        get_response = api_client.get(f"{API_BASE_URL}/users/{user_id}", headers=auth_headers)
        assert get_response.status_code == 404


class TestAuthenticationAPI:
    """Test suite for authentication endpoints"""

    def test_login_success(self, api_client):
        """Test successful login returns token"""
        response = api_client.post(
            f"{API_BASE_URL}/auth/login",
            json={"username": "test_user", "password": "Test123!"}
        )

        assert response.status_code == 200
        data = response.json()
        assert "token" in data
        assert len(data["token"]) > 20  # Token should be substantial

    def test_login_invalid_credentials(self, api_client):
        """Test login with invalid credentials returns 401"""
        response = api_client.post(
            f"{API_BASE_URL}/auth/login",
            json={"username": "test_user", "password": "WrongPassword"}
        )

        assert response.status_code == 401
        assert "invalid" in response.json()["error"].lower()

    def test_protected_endpoint_without_auth(self, api_client):
        """Test accessing protected endpoint without token returns 401"""
        response = api_client.get(f"{API_BASE_URL}/users/me")

        assert response.status_code == 401


class TestResponseTimes:
    """Test suite for API performance"""

    def test_api_response_time_under_sla(self, api_client, auth_headers):
        """Test API responds within SLA (500ms)"""
        import time

        start = time.time()
        response = api_client.get(f"{API_BASE_URL}/users", headers=auth_headers)
        duration = (time.time() - start) * 1000  # Convert to ms

        assert response.status_code == 200
        assert duration < 500, f"Response took {duration}ms, SLA is 500ms"

<best_practices>
1. Use AAA pattern: Arrange, Act, Assert
2. Test all HTTP methods: GET, POST, PUT, PATCH, DELETE
3. Validate status codes AND response bodies
4. Test authentication and authorization scenarios
5. Test error cases (400, 401, 403, 404, 500)
6. Use unique test data to avoid conflicts
7. Clean up test data after tests (or use transactions)
8. Validate response times for performance SLAs
9. Test pagination, filtering, sorting if applicable
10. Use fixtures for auth tokens and API clients
</best_practices>
