```python
"""
Mobile Test Suite using Appium
Tests iOS and Android native apps
"""

from appium import webdriver
from appium.webdriver.common.appiumby import AppiumBy
import pytest


@pytest.fixture
def android_driver():
    """Fixture for Android driver"""
    caps = {
        "platformName": "Android",
        "deviceName": "Android Emulator",
        "app": "/path/to/app.apk",
        "automationName": "UiAutomator2"
    }
    driver = webdriver.Remote("http://localhost:4723/wd/hub", caps)
    yield driver
    driver.quit()


class TestMobileApp:
    def test_app_launch(self, android_driver):
        """Test app launches successfully"""
        assert android_driver.current_activity == ".MainActivity"

    def test_login_flow(self, android_driver):
        """Test mobile login flow"""
        # Find and interact with elements
        username = android_driver.find_element(AppiumBy.ID, "username")
        username.send_keys("testuser")

        password = android_driver.find_element(AppiumBy.ID, "password")
        password.send_keys("password123")

        login_btn = android_driver.find_element(AppiumBy.ID, "loginButton")
        login_btn.click()

        # Verify login success
        welcome = android_driver.find_element(AppiumBy.ID, "welcomeMessage")
        assert "Welcome" in welcome.text
```
