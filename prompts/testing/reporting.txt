"""
Test Reporting Configuration
Generates comprehensive test reports using pytest-html and Allure
"""

import pytest
from datetime import datetime


# pytest configuration for reporting
def pytest_configure(config):
    """Configure pytest reporting"""
    config.option.htmlpath = f"reports/test-report-{datetime.now().strftime('%Y%m%d-%H%M%S')}.html"
    config.option.self_contained_html = True


@pytest.hookimpl(hookwrapper=True)
def pytest_runtest_makereport(item, call):
    """Capture screenshots on test failure"""
    outcome = yield
    report = outcome.get_result()

    if report.when == "call" and report.failed:
        # Capture screenshot if test failed
        if hasattr(item, "funcargs") and "page" in item.funcargs:
            page = item.funcargs["page"]
            screenshot_path = f"screenshots/failure_{item.name}.png"
            page.screenshot(path=screenshot_path)
            report.extra = [("Screenshot", screenshot_path)]


# Allure reporting decorators
@pytest.mark.allure.feature("Authentication")
@pytest.mark.allure.story("User Login")
def test_with_allure_metadata():
    """Test with Allure metadata for rich reporting"""
    pass
