```python
"""
UI Test Suite for {{WEB_URL}}
Generated by ROUGE - Test Automation Platform

Test Coverage:
- Authentication flows (login, logout, signup)
- Critical user journeys
- Form validations
- Navigation
- Error handling
"""

import pytest
from playwright.sync_api import Page, expect


# ============================================
# Page Object Models
# ============================================

class LoginPage:
    """Page Object for login page"""

    def __init__(self, page: Page):
        self.page = page
        self.username_input = page.locator("#username")
        self.password_input = page.locator("#password")
        self.login_button = page.locator("button[type='submit']")
        self.error_message = page.locator(".error-message")

    def navigate(self):
        """Navigate to login page"""
        self.page.goto("{{WEB_URL}}/login")
        expect(self.page).to_have_url("{{WEB_URL}}/login")

    def login(self, username: str, password: str):
        """Perform login action"""
        self.username_input.fill(username)
        self.password_input.fill(password)
        self.login_button.click()

    def get_error_message(self) -> str:
        """Get error message text"""
        return self.error_message.inner_text()


class DashboardPage:
    """Page Object for dashboard"""

    def __init__(self, page: Page):
        self.page = page
        self.welcome_message = page.locator("h1")
        self.logout_button = page.locator("button:has-text('Logout')")

    def is_loaded(self) -> bool:
        """Check if dashboard is loaded"""
        return self.welcome_message.is_visible()

    def logout(self):
        """Perform logout action"""
        self.logout_button.click()


# ============================================
# Test Fixtures
# ============================================

@pytest.fixture(scope="function")
def login_page(page: Page) -> LoginPage:
    """Fixture to provide LoginPage instance"""
    return LoginPage(page)


@pytest.fixture(scope="function")
def dashboard_page(page: Page) -> DashboardPage:
    """Fixture to provide DashboardPage instance"""
    return DashboardPage(page)


@pytest.fixture(scope="function")
def test_user():
    """Fixture to provide test user credentials"""
    return {
        "username": "test_user",
        "password": "Test123!",
        "email": "test@example.com"
    }


# ============================================
# Test Suites
# ============================================

class TestAuthentication:
    """Test suite for authentication flows"""

    def test_successful_login(self, login_page, dashboard_page, test_user):
        """Test successful login with valid credentials"""
        # Arrange
        login_page.navigate()

        # Act
        login_page.login(test_user["username"], test_user["password"])

        # Assert
        assert dashboard_page.is_loaded()
        expect(dashboard_page.welcome_message).to_contain_text(test_user["username"])

    def test_login_with_invalid_credentials(self, login_page, test_user):
        """Test login fails with invalid credentials"""
        # Arrange
        login_page.navigate()

        # Act
        login_page.login(test_user["username"], "wrong_password")

        # Assert
        error_message = login_page.get_error_message()
        assert "Invalid credentials" in error_message or "Login failed" in error_message

    def test_logout_flow(self, login_page, dashboard_page, test_user):
        """Test user can logout successfully"""
        # Arrange - login first
        login_page.navigate()
        login_page.login(test_user["username"], test_user["password"])
        assert dashboard_page.is_loaded()

        # Act
        dashboard_page.logout()

        # Assert - should redirect to login
        expect(login_page.page).to_have_url("{{WEB_URL}}/login")


class TestNavigation:
    """Test suite for navigation flows"""

    def test_main_navigation_links(self, page: Page):
        """Test all main navigation links are working"""
        page.goto("{{WEB_URL}}")

        # Test each navigation link
        navigation_items = [
            ("Home", "{{WEB_URL}}/"),
            ("About", "{{WEB_URL}}/about"),
            ("Contact", "{{WEB_URL}}/contact"),
        ]

        for link_text, expected_url in navigation_items:
            link = page.locator(f"nav a:has-text('{link_text}')")
            link.click()
            expect(page).to_have_url(expected_url)
            page.go_back()


# Add more test classes for:
# - Form validations
# - Error handling
# - Critical user journeys
# - Responsive design (if needed)
```

<best_practices>
1. Use Page Object Model to separate page structure from test logic
2. Use explicit waits (expect) not arbitrary sleeps
3. Use meaningful locators (data-testid preferred, then accessible labels)
4. One test = one assertion concept (when possible)
5. Tests must be independent (can run in any order)
6. Use fixtures for setup/teardown, not before/after hooks
7. Clear test names that describe behavior: test_user_can_login_with_valid_credentials
8. AAA pattern: Arrange, Act, Assert
9. Handle dynamic content with proper waits
10. Take screenshots on failure for debugging
</best_practices>
