```python
"""
Accessibility Test Suite
Validates WCAG 2.1 AA compliance using axe-core
"""

import pytest
from playwright.sync_api import Page
from axe_playwright_python.sync_playwright import Axe


class TestAccessibility:
    def test_homepage_accessibility(self, page: Page):
        """Test homepage for accessibility violations"""
        page.goto("{{WEB_URL}}")

        axe = Axe()
        results = axe.run(page)

        # Assert no violations
        violations = results.violations
        assert len(violations) == 0, f"Found {len(violations)} accessibility violations"

    def test_keyboard_navigation(self, page: Page):
        """Test keyboard navigation works"""
        page.goto("{{WEB_URL}}")

        # Tab through interactive elements
        page.keyboard.press("Tab")
        focused_element = page.evaluate("document.activeElement.tagName")
        assert focused_element in ["A", "BUTTON", "INPUT"]

    def test_color_contrast(self, page: Page):
        """Test color contrast meets WCAG standards"""
        page.goto("{{WEB_URL}}")

        axe = Axe()
        results = axe.run(page, {"rules": {"color-contrast": {"enabled": True}}})

        contrast_violations = [v for v in results.violations if v["id"] == "color-contrast"]
        assert len(contrast_violations) == 0
```
