"""
Test Data Factories
Generates realistic test data using Faker
"""

from faker import Faker
import pytest
from datetime import datetime

fake = Faker()


class UserFactory:
    """Factory for generating user test data"""

    @staticmethod
    def build(**kwargs):
        """Build a user dict with fake data"""
        user = {
            "id": fake.uuid4(),
            "username": fake.user_name(),
            "email": fake.email(),
            "first_name": fake.first_name(),
            "last_name": fake.last_name(),
            "password": "Test123!",
            "created_at": fake.date_time_this_year().isoformat(),
        }
        user.update(kwargs)  # Override with custom values
        return user

    @staticmethod
    def build_batch(count=5, **kwargs):
        """Build multiple users"""
        return [UserFactory.build(**kwargs) for _ in range(count)]


# Add factories for other entities (Product, Order, etc.)


# Pytest fixtures
@pytest.fixture
def test_user():
    """Fixture providing a test user"""
    return UserFactory.build()


@pytest.fixture
def test_users():
    """Fixture providing multiple test users"""
    return UserFactory.build_batch(count=3)
